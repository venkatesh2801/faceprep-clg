package sample;
import java.util.Scanner;

public class AbsoluteValue {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter a number: ");
        int num = sc.nextInt();
        
        int absValue;
        if (num < 0) {
            absValue = -num;  
        } else {
            absValue = num;  
        }
        System.out.println("Absolute value: " + absValue);
        sc.close();
    }
}
